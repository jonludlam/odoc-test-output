Module type Monad_intf.Infix3

Same as Infix, except the monad type has three arguments. The second and third are always just passed through.

type ('a, 'd, 'e) t
val (>>=) : ('a, 'd, 'e) t -> ('a -> ('b, 'd, 'e) t) -> ('b, 'd, 'e) t
val (>>|) : ('a, 'd, 'e) t -> ('a -> 'b) -> ('b, 'd, 'e) t
